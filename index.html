<head>

    <title>Live Shipment Tracking</title>

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <link rel="stylesheet" href="https://unpkg.com/leaflet@1.7.1/dist/leaflet.css" />

    <style>

        body { margin: 0; padding: 0; }

        #map { height: 100vh; width: 100%; }

    </style>

</head>

<body>


    <div id="map"></div>


    <script src="https://unpkg.com/leaflet@1.7.1/dist/leaflet.js"></script>

    <script>

        // 1. Setup the Map

        var map = L.map('map').setView([0, 0], 2); // Starts zoomed out

        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

            attribution: '© OpenStreetMap contributors'

        }).addTo(map);


        // Add a car marker

        var carIcon = L.icon({

            iconUrl: 'https://cdn-icons-png.flaticon.com/512/741/741407.png', // Simple car icon

            iconSize: [40, 40],

        });

        var marker = L.marker([0, 0], {icon: carIcon}).addTo(map);


        // 2. Get IMEI from the URL (e.g., website.com?imei=12345)

        const urlParams = new URLSearchParams(window.location.search);

        const imei = urlParams.get('imei');


        if (!imei) {

            alert("No Tracking ID provided!");

        }


        // 3. Function to fetch data from Make

        async function updateLocation() {

            if(!imei) return;


            try {

                // Call Make Webhook with the IMEI

                const response = await fetch(`PASTE_YOUR_MAKE_WEBHOOK_URL_HERE?imei=${imei}`);

                const data = await response.json();


                if (data.lat && data.lng) {

                    var newLatLng = new L.LatLng(data.lat, data.lng);

                    marker.setLatLng(newLatLng);

                    map.panTo(newLatLng); // Center map on car

                    map.setZoom(15);      // Zoom in

                }

            } catch (error) {

                console.error("Error fetching location:", error);

            }

        }


        // Run immediately, then every 5 seconds

        updateLocation();

        setInterval(updateLocation, 5000);


    </script>

</body>

</html>
